替换目录中指定后缀名文件中的内容

发表于 2015-07-01

昨天由于不会玩Hexo，点击很多url会404，费了好多时间都没弄好。但是自己还是写了一点自认为有用的代码，现在贴在这里：

import java.io.BufferedReader;
import java.io.File;
import java.io.FileInputStream;
import java.io.FileOutputStream;
import java.io.IOException;
import java.io.InputStreamReader;
import java.io.OutputStreamWriter;
import java.io.PrintWriter;
import java.util.ArrayList;


public class RepalceHexo {
    public static void main(String[] args) {

        ArrayList<File> al = getFile(new File("D:\\Hexo\\public"));

        for(File f : al){
            if (f.getName().matches(".*\\.html$")){ // 匹配java格式的文件
                replaceFile(f);
            }                        
        }

    private static void replaceFile(File f) {
        BufferedReader br = null;
        PrintWriter bw =null;
        try {
            br = new BufferedReader(
                new InputStreamReader(
                    new FileInputStream(f),"utf-8"));            
            bw = new PrintWriter(
                new OutputStreamWriter(
                    new FileOutputStream(
                        new File(f.getAbsoluteFile()+"_")),"utf-8"));
            String line = "";
            String s = "";
            String s1 = "";
            while ((line = br.readLine()) != null) {
                line = line.trim();

//                s= line.replaceAll("href=\"/mybolg/", "href=\"/");
//                s1 = line.replaceAll("href=\"/mybolg/", "src=\"/");
                s= line.replaceAll("href=\"/", "href=\"/mybolg/");                    
                s1 = line.replaceAll("src=\"/", "href=\"/mybolg/");                    
                if(s!=line){
                    bw.write(s);
                }else if(s1!=line){
                    bw.write(s1);                    
                }else{
                    bw.write(line);                        
                }        
            }
//            br.close();
//            bw.close();
//            br = new BufferedReader(
//                    new InputStreamReader(
//                        new FileInputStream(f.getAbsoluteFile()+"_"),"utf-8"));            
//            bw = new PrintWriter(
//                    new OutputStreamWriter(
//                        new FileOutputStream(
//                            new File(f.getAbsoluteFile()+"")),"utf-8"));
//            
//            while ((line = br.readLine()) != null) {
//                line = line.trim();            
//                bw.write(line);                                                
//            }

        } catch (Exception e) {
            e.printStackTrace();
        } finally {
            if (br != null) {
                try {
                    br.close();
                    br = null;
                } catch (IOException e) {
                    e.printStackTrace();
                }
            }
            if (bw != null) {
                bw.close();
                bw = null;
            }
        }
    }
    //递归取得指定目录下的全部文件
    static ArrayList<File> fileArray = new ArrayList<File>();
    public static ArrayList<File> getFile(File f) {
        File[] ff = f.listFiles();
        for (File child : ff) {
            if (child.isDirectory()) {
                getFile(child);
            } else
                fileArray.add(child);            
        }
        return fileArray;

    }
}
